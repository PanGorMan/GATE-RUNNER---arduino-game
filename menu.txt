#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <avr/wdt.h>

const int pin2 = 2; // góra 
const int pin3 = 3; // wybór
const int pin5 = 5; // dół

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1

int stan = 1 ;
bool selected = false ;

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

void setup() {

  pinMode(pin2, INPUT_PULLUP);
  pinMode(pin3, INPUT_PULLUP);
  pinMode(pin5, INPUT_PULLUP);

  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
}

void loop() {

if (!selected) {

  if (digitalRead(pin2) == LOW) {
    stan--;
    delay(150);
  }

  if (digitalRead(pin5) == LOW) {
    stan++;
    delay(150);
  }

}

  if (digitalRead(pin3) == LOW) {
    selected = !selected;
    delay(150);
  }

  // ograniczenie menu
  if (stan < 1) stan = 1;
  if (stan > 3) stan = 3;

  display.clearDisplay();
  display.setTextSize(2);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);

  if (!selected) {

    display.print("TITLE\n");

    if (stan == 1) display.print(">"); else display.print(" ");
    display.print("START\n");

    if (stan == 2) display.print(">"); else display.print(" ");
    display.print("RULES\n");

    if (stan == 3) display.print(">"); else display.print(" ");
    display.print("QUIT\n");

  }
else {

  if (stan == 1){
    display.print("QWE QWE QWE\n");
  }

  if (stan == 2){
    display.setTextSize(1);
    display.print("Go through left to right and back.\nAvoid square if you don't want to lose.\nTo win earn 2000 points.\nGood luck : )\n");
  }

  if (stan == 3){

    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 0);
    display.print("Game will turn shut down \nPress any button to  turn it on ");

    display.display();   // ← TO MUSI BYĆ TU

    delay(3000);

    display.clearDisplay();
    display.display();

    while (digitalRead(pin2) == HIGH &&
           digitalRead(pin3) == HIGH &&
           digitalRead(pin5) == HIGH) {
    }
    wdt_enable(WDTO_15MS);

    while (1) {}
  }
}

  display.display();
}
